class Base(DeclarativeBase):
    pass

class User(Base):
    __tablename__ = 'users'
    id = Column(Integer, primary_key=True)
    telegram_id = Column(Integer, unique=True)
    username = Column(String)
    balance = Column(Float, default=0)
    total_spent = Column(Float, default=0)
    is_admin = Column(Boolean, default=False)
    is_banned = Column(Boolean, default=False)
    created_at = Column(DateTime, default=datetime.now)
    last_activity = Column(DateTime, default=datetime.now)

class Purchase(Base):
    __tablename__ = 'purchases'
    id = Column(Integer, primary_key=True)
    user_id = Column(Integer)
    service = Column(String)
    country = Column(String)
    phone = Column(String)
    sms_code = Column(String)
    price_rub = Column(Float)
    group_id = Column(Integer, nullable=True)
    group_owner_id = Column(Integer, nullable=True)
    status = Column(String, default='completed')
    created_at = Column(DateTime, default=datetime.now)

class Group(Base):
    __tablename__ = 'groups'
    id = Column(Integer, primary_key=True)
    group_id = Column(Integer, unique=True)
    group_username = Column(String, nullable=True)
    owner_id = Column(Integer)
    is_active = Column(Boolean, default=True)
    created_at = Column(DateTime, default=datetime.now)

class Transaction(Base):
    __tablename__ = 'transactions'
    id = Column(Integer, primary_key=True)
    user_id = Column(Integer)
    type = Column(String)
    amount = Column(Float)
    description = Column(String)
    invoice_id = Column(Integer, nullable=True)
    created_at = Column(DateTime, default=datetime.now)

class SupportTicket(Base):
    __tablename__ = 'support_tickets'
    id = Column(Integer, primary_key=True)
    user_id = Column(Integer)
    message = Column(String)
    status = Column(String, default='open')
    created_at = Column(DateTime, default=datetime.now)

engine = create_async_engine('sqlite+aiosqlite:///bot.db')
async_session = async_sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)

async def init_db():
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)

class SimService:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://5sim.net/v1'
    
    async def get_balance(self) -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Authorization': f'Bearer {self.api_key}', 'Accept': 'application/json'}
            async with session.get(f'{self.base_url}/user/profile', headers=headers) as resp:
                return await resp.json()
    
    async def get_prices(self, country: str = 'any') -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Authorization': f'Bearer {self.api_key}', 'Accept': 'application/json'}
            params = {'country': country}
            async with session.get(f'{self.base_url}/guest/prices', headers=headers, params=params) as resp:
                return await resp.json()
    
    async def buy_number(self, country: str, product: str = 'imo') -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Authorization': f'Bearer {self.api_key}', 'Accept': 'application/json'}
            async with session.get(f'{self.base_url}/user/buy/activation/{country}/any/{product}', headers=headers) as resp:
                return await resp.json()
    
    async def check_order(self, order_id: str) -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Authorization': f'Bearer {self.api_key}', 'Accept': 'application/json'}
            async with session.get(f'{self.base_url}/user/check/{order_id}', headers=headers) as resp:
                return await resp.json()
    
    async def cancel_order(self, order_id: str) -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Authorization': f'Bearer {self.api_key}', 'Accept': 'application/json'}
            async with session.get(f'{self.base_url}/user/cancel/{order_id}', headers=headers) as resp:
                return await resp.json()

class CryptoService:
    def __init__(self):
        self.api_key = CRYPTOBOT_API_KEY
        self.base_url = 'https://pay.crypt.bot/api'
    
    async def create_invoice(self, amount_rub: float, user_id: int) -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Crypto-Pay-API-Token': self.api_key}
            data = {
                'asset': 'USDT',
                'amount': str(amount_rub),
                'description': 'Пополнение баланса IMO Shop',
                'payload': str(user_id),
                'allow_anonymous': False,
                'fiat': 'RUB'
            }
            async with session.post(f'{self.base_url}/createInvoice', headers=headers, json=data) as resp:
                return await resp.json()
    
    async def get_invoice(self, invoice_id: int) -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Crypto-Pay-API-Token': self.api_key}
            async with session.get(f'{self.base_url}/getInvoices?invoice_ids={invoice_id}', headers=headers) as resp:
                result = await resp.json()
                if 'items' in result and len(result['items']) > 0:
                    return result['items'][0]
                return result

@user_router.message(UserStates.waiting_for_custom_country)
async def process_custom_country(message: Message, state: FSMContext, session: AsyncSession):
    country = message.text.strip().lower()
    
    if country in EXCLUDED_COUNTRIES:
        await message.answer("❌ Эта страна недоступна.", reply_markup=back_button("buy_imo"))
        await state.clear()
        return
    
    await message.answer("🔍 Ищу лучшую цену...")
    
    try:
        sim_prices = await sim_service.get_prices()
        price_rub = None
        service = None
        
        if country in sim_prices.get('prices', {}) and 'imo' in sim_prices['prices'][country]:
            price_rub = round(float(sim_prices['prices'][country]['imo']['cost']) * USD_RATE * 1.2, 2)
            service = '5sim'
        
        if price_rub:
            await message.answer(
                f"🔍 Найдено:\n\n🌍 {country.capitalize()}\n💰 {price_rub}₽/шт\n🔧 {service}",
                reply_markup=get_quantity_keyboard(country, price_rub, service)
            )
        else:
            await message.answer("❌ Нет номеров.", reply_markup=back_button("buy_imo"))
    except Exception as e:
        await message.answer(f"❌ Ошибка: {str(e)}", reply_markup=back_button("buy_imo"))
    
    await state.clear()

@user_router.callback_query(F.data.startswith("country_"))
async def select_country(callback: CallbackQuery):
    country = callback.data.replace("country_", "")
    
    if country == "custom":
        return
    
    await callback.message.edit_text("🔍 Ищу лучшую цену...")
    
    try:
        price_rub = None
        service = None
        country_name = country
        
        sim_prices = await sim_service.get_prices()
        
        if country == "cheapest":
            best_price = float('inf')
            for c, products in sim_prices.get('prices', {}).items():
                if c.lower() in EXCLUDED_COUNTRIES:
                    continue
                if 'imo' in products:
                    price = float(products['imo']['cost']) * USD_RATE * 1.2
                    if price < best_price:
                        best_price = price
                        country_name = c
            if best_price != float('inf'):
                price_rub = round(best_price, 2)
                service = '5sim'
        else:
            if country in sim_prices.get('prices', {}) and 'imo' in sim_prices['prices'][country]:
                price_rub = round(float(sim_prices['prices'][country]['imo']['cost']) * USD_RATE * 1.2, 2)
                service = '5sim'
        
        if price_rub:
            await callback.message.edit_text(
                f"🔍 Найдено:\n\n🌍 {country_name.capitalize()}\n💰 {price_rub}₽/шт\n🔧 {service}\n\nВыберите количество:",
                reply_markup=get_quantity_keyboard(country_name, price_rub, service)
            )
        else:
            await callback.message.edit_text("❌ Нет номеров.", reply_markup=back_button("buy_imo"))
    except Exception as e:
        await callback.message.edit_text(f"❌ Ошибка: {str(e)}", reply_markup=back_button("buy_imo"))
    await callback.answer()

@user_router.callback_query(F.data.startswith("qty_"))
async def process_quantity(callback: CallbackQuery, session: AsyncSession):
    _, qty, service, country, price_per_one = callback.data.split("_")
    qty = int(qty)
    price_per_one = float(price_per_one)
    total_price = round(price_per_one * qty, 2)
    
    user = await session.get(User, callback.from_user.id)
    
    if user.balance < total_price:
        await callback.message.edit_text(
            f"❌ Недостаточно средств!\n\nКоличество: {qty} шт\nЦена за шт: {price_per_one}₽\nИтого: {total_price}₽\nБаланс: {user.balance:.2f}₽",
            reply_markup=back_button(f"country_{country}")
        )
        await callback.answer()
        return
    
    await callback.message.edit_text(f"🔄 Покупаю {qty} шт IMO...\n\n0/{qty}")
    
    success_list = []
    failed = 0
    
    for i in range(qty):
        try:
            result = await sim_service.buy_number(country)
            
            if result.get('id'):
                order_id = str(result['id'])
                phone = result.get('phone', '')
                
                sms_received = False
                for _ in range(30):
                    await asyncio.sleep(2)
                    check = await sim_service.check_order(order_id)
                    
                    if check.get('status') == 'RECEIVED':
                        sms_code = check.get('sms', '') or check.get('sms_text', '')
                        
                        user.balance -= price_per_one
                        user.total_spent += price_per_one
                        user.last_activity = datetime.now()
                        
                        purchase = Purchase(
                            user_id=user.telegram_id, service=service,
                            country=country, phone=phone,
                            sms_code=str(sms_code), price_rub=price_per_one
                        )
                        session.add(purchase)
                        session.add(Transaction(
                            user_id=user.telegram_id, type='purchase',
                            amount=-price_per_one, description=f'IMO {country}'
                        ))
                        await session.commit()
                        
                        success_list.append(f"📱 {phone} | 💬 {sms_code}")
                        sms_received = True
                        break
                
                if not sms_received:
                    await sim_service.cancel_order(order_id)
                    failed += 1
            else:
                failed += 1
        except:
            failed += 1
        
        await callback.message.edit_text(f"🔄 Покупаю {qty} шт IMO...\n\n{i+1}/{qty}\n✅ {len(success_list)} | ❌ {failed}")
    
    text = f"✅ Закупка завершена!\n\nКуплено: {len(success_list)}/{qty}\nЦена за шт: {price_per_one}₽\nПотрачено: {round(price_per_one * len(success_list), 2)}₽\n💰 Баланс: {user.balance:.2f}₽\n\n"
    
    if success_list:
        text += "Аккаунты:\n"
        for s in success_list[:10]:
            text += f"{s}\n"
        if len(success_list) > 10:
            text += f"...и ещё {len(success_list) - 10}"
    
    builder = InlineKeyboardBuilder()
    builder.button(text="🛒 Купить ещё", callback_data="buy_imo")
    builder.button(text="🔙 В меню", callback_data="main_menu")
    builder.adjust(1)
    
    await callback.message.edit_text(text, reply_markup=builder.as_markup())
    await callback.answer()

        f"💸 Потрачено: {user.total_spent:.2f}₽\n"

    if purchases:
        text += "\n\nПоследние покупки:\n"

@group_router.callback_query(F.data.startswith("g_"))
async def group_purchase(callback: CallbackQuery, session: AsyncSession):
    parts = callback.data.split("_")
    country = parts[1]
    group_id = int(parts[2])
    
    group = await session.get(Group, group_id)
    if not group:
        await callback.answer("Группа не подключена.")
        return
    
    owner = await session.get(User, group.owner_id)
    if not owner:
        await callback.answer("Владелец не найден.")
        return
    
    try:
        sim_prices = await sim_service.get_prices()
        price_rub = None
        country_name = country
        
        if country == "cheapest":
            best_price = float('inf')
            for c, products in sim_prices.get('prices', {}).items():
                if c.lower() in EXCLUDED_COUNTRIES:
                    continue
                if 'imo' in products:
                    price = float(products['imo']['cost']) * USD_RATE * 1.2
                    if price < best_price:
                        best_price = price
                        country_name = c
            if best_price != float('inf'):
                price_rub = round(best_price, 2)
        else:
            if country in sim_prices.get('prices', {}) and 'imo' in sim_prices['prices'][country]:
                price_rub = round(float(sim_prices['prices'][country]['imo']['cost']) * USD_RATE * 1.2, 2)
        
        if not price_rub:
            await callback.answer("Нет номеров.")
            return
        
        if owner.balance < price_rub:
            await callback.answer(f"❌ У владельца недостаточно средств. Нужно {price_rub}₽.")
            return
        
        result = await sim_service.buy_number(country_name)
        
        if result.get('id'):
            order_id = str(result['id'])
            phone = result.get('phone', '')
            
            for _ in range(30):
                await asyncio.sleep(2)
                check = await sim_service.check_order(order_id)
                
                if check.get('status') == 'RECEIVED':
                    sms_code = check.get('sms', '') or check.get('sms_text', '')
                    
                    owner.balance -= price_rub
                    owner.total_spent += price_rub
                    owner.last_activity = datetime.now()
                    
                    purchase = Purchase(
                        user_id=callback.from_user.id, service='5sim',
                        country=country_name, phone=phone,
                        sms_code=str(sms_code), price_rub=price_rub,
                        group_id=group_id, group_owner_id=owner.telegram_id
                    )
                    session.add(purchase)
                    session.add(Transaction(
                        user_id=owner.telegram_id, type='purchase',
                        amount=-price_rub, description=f'IMO {country_name} (группа)'
                    ))
                    await session.commit()
                    
                    await callback.message.edit_text(
                        f"✅ Готово!\n\n📱 {phone}\n🌍 {country_name.capitalize()}\n💬 Код: {sms_code}\n💰 Списано с владельца: {price_rub}₽"
                    )
                    await callback.answer()
                    return
            
            await sim_service.cancel_order(order_id)
            await callback.answer("❌ SMS не получена.")
        else:
            await callback.answer("❌ Ошибка покупки.")
    except Exception as e:
        await callback.answer(f"Ошибка: {str(e)}")

        await callback.message.edit_text("👥 Нет подключённых групп.", reply_markup=back_button("admin_back"))
        await callback.answer()
        return
    
    text = "👥 Подключённые группы:\n\n"
    for g in groups:
        user = await session.get(User, g.owner_id)
        username = f"@{user.username}" if user and user.username else g.owner_id
        text += f"ID: {g.group_id}\nВладелец: {username}\n\n"
    
    await callback.message.edit_text(text, reply_markup=back_button("admin_back"))
    await callback.answer()

@admin_router.callback_query(F.data == "admin_usd_rate")
async def admin_usd_rate(callback: CallbackQuery, state: FSMContext):
    if not is_admin(callback.from_user.id):
        return
    await callback.message.edit_text(f"💱 Текущий курс USD: {USD_RATE}₽\n\nВведите новый курс:")
    await state.set_state(AdminStates.waiting_for_usd_rate)
    await callback.answer()

@admin_router.message(AdminStates.waiting_for_usd_rate)
async def process_usd_rate(message: Message, state: FSMContext):
    global USD_RATE
    if not is_admin(message.from_user.id):
        return
    
    try:
        new_rate = float(message.text.strip())
        USD_RATE = new_rate
        await message.answer(f"✅ Курс USD обновлён: {USD_RATE}₽")
    except ValueError:
        await message.answer("❌ Неверное значение. Введите число.")
    
    await state.clear()

@admin_router.callback_query(F.data == "admin_back")
async def admin_back(callback: CallbackQuery):
    if not is_admin(callback.from_user.id):
        return
    await callback.message.edit_text("🔧 Админ-панель", reply_markup=get_admin_keyboard())
    await callback.answer()

@admin_router.callback_query(F.data == "admin_exit")
async def admin_exit(callback: CallbackQuery):
    await callback.message.delete()
    await callback.answer()
