class SmsFastService:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://smsfastapi.com/stubs/handler_api.php'
    
    async def get_balance(self) -> str:
        async with aiohttp.ClientSession() as session:
            params = {'api_key': self.api_key, 'action': 'getBalance'}
            async with session.get(self.base_url, params=params) as resp:
                text = await resp.text()
                if 'ACCESS_BALANCE:' in text:
                    return text.split(':')[1]
                return '0'
    
    async def get_prices(self, country: str, service: str = 'imo') -> Dict:
        async with aiohttp.ClientSession() as session:
            params = {'api_key': self.api_key, 'action': 'getPrices', 'service': service, 'country': country}
            async with session.get(self.base_url, params=params) as resp:
                text = await resp.text()
                try:
                    return await resp.json()
                except:
                    return {}
    
    async def get_all_prices(self) -> list:
        """Собрать цены по всем странам"""
        result = []
        for country in COUNTRIES:
            try:
                prices = await self.get_prices(country)
                if prices and country in prices and 'imo' in prices[country]:
                    imo_data = prices[country]['imo']
                    if isinstance(imo_data, dict):
                        for operator, price in imo_data.items():
                            result.append({'country': country, 'operator': operator, 'price': float(price)})
                    else:
                        result.append({'country': country, 'operator': 'any', 'price': float(imo_data)})
            except:
                pass
        return result
    
    async def buy_number(self, service: str = 'imo', country: str = 'any') -> Dict:
        async with aiohttp.ClientSession() as session:
            params = {'api_key': self.api_key, 'action': 'getNumber', 'service': service, 'country': country}
            async with session.get(self.base_url, params=params) as resp:
                text = await resp.text()
                if 'ACCESS_NUMBER:' in text:
                    parts = text.split(':')
                    return {'id': parts[1], 'phone': parts[2]}
                return {'error': text}
    
    async def get_sms(self, activation_id: str) -> str:
        async with aiohttp.ClientSession() as session:
            params = {'api_key': self.api_key, 'action': 'getStatus', 'id': activation_id}
            async with session.get(self.base_url, params=params) as resp:
                text = await resp.text()
                if 'STATUS_OK:' in text:
                    return text.split(':')[1]
                return ''
    
    async def cancel_order(self, activation_id: str):
        async with aiohttp.ClientSession() as session:
            params = {'api_key': self.api_key, 'action': 'setStatus', 'status': '8', 'id': activation_id}
            async with session.get(self.base_url, params=params) as resp:
                return await resp.text()

class CryptoService:
    def __init__(self):
        self.api_key = CRYPTOBOT_API_KEY
        self.base_url = 'https://pay.crypt.bot/api'
    
    async def create_invoice(self, amount_rub: float, user_id: int) -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Crypto-Pay-API-Token': self.api_key}
            data = {
                'asset': 'USDT',
                'amount': str(amount_rub),
                'description': 'Пополнение IMO Shop',
                'payload': str(user_id),
                'allow_anonymous': False,
                'fiat': 'RUB'
            }
            async with session.post(f'{self.base_url}/createInvoice', headers=headers, json=data) as resp:
                return await resp.json()
    
    async def get_invoice(self, invoice_id: int) -> Dict:
        async with aiohttp.ClientSession() as session:
            headers = {'Crypto-Pay-API-Token': self.api_key}
            async with session.get(f'{self.base_url}/getInvoices?invoice_ids={invoice_id}', headers=headers) as resp:
                result = await resp.json()
                if 'items' in result and len(result['items']) > 0:
                    return result['items'][0]
                return result

    if country == "custom":
        return
    
    await callback.message.edit_text("🔍 Ищу...")
    
    try:
        prices = await get_prices_cached()
        
        price_rub = None
        found_country = country
        
        if country == "any":
            best = min(prices, key=lambda x: x['price']) if prices else None
            if best:
                price_rub = round(best['price'] * get_markup(), 2)
                found_country = best['country']
        else:
            found = [p for p in prices if p['country'] == country]
            if found:
                best = min(found, key=lambda x: x['price'])
                price_rub = round(best['price'] * get_markup(), 2)
        
        if price_rub:
            await callback.message.edit_text(
                f"🔍 {found_country.upper()}\n💰 {price_rub}₽/шт\nНаценка: {MARKUP_PERCENT}%\n\nВыберите количество:",
                reply_markup=get_quantity_keyboard(found_country, price_rub)
            )
        else:
            await callback.message.edit_text("❌ Нет номеров.", reply_markup=back_button("buy_imo"))
    except Exception as e:
        await callback.message.edit_text(f"❌ {str(e)}", reply_markup=back_button("buy_imo"))
    await callback.answer()

@user_router.callback_query(F.data.startswith("qty_"))
async def process_quantity(callback: CallbackQuery, session: AsyncSession):
    _, qty, country, price_per_one = callback.data.split("_")
    qty = int(qty)
    price_per_one = float(price_per_one)
    total_price = round(price_per_one * qty, 2)
    
    user = await session.get(User, callback.from_user.id)
    
    if user.balance < total_price:
        await callback.message.edit_text(
            f"❌ Недостаточно средств!\n{qty} шт × {price_per_one}₽ = {total_price}₽\nБаланс: {user.balance:.2f}₽",
            reply_markup=back_button(f"country_{country}")
        )
        await callback.answer()
        return
    
    await callback.message.edit_text(f"🔄 Покупаю {qty} шт...\n0/{qty}")
    
    success = []
    failed = 0
    
    for i in range(qty):
        try:
            result = await smsfast_service.buy_number(country=country)
            if result.get('id') and 'error' not in result:
                order_id = result['id']
                phone = result['phone']
                
                for _ in range(30):
                    await asyncio.sleep(2)
                    code = await smsfast_service.get_sms(order_id)
                    if code:
                        user.balance -= price_per_one
                        user.total_spent += price_per_one
                        user.last_activity = datetime.now()
                        
                        session.add(Purchase(user_id=user.telegram_id, country=country, phone=phone, sms_code=code, price_rub=price_per_one))
                        session.add(Transaction(user_id=user.telegram_id, type='purchase', amount=-price_per_one, description=f'IMO {country}'))
                        await session.commit()
                        
                        success.append(f"📱 {phone} | 💬 {code}")
                        break
                else:
                    try:
                        await smsfast_service.cancel_order(order_id)
                    except:
                        pass
                    failed += 1
            else:
                failed += 1
        except:
            failed += 1
        
        await callback.message.edit_text(f"🔄 Покупаю {qty} шт...\n{i+1}/{qty}\n✅ {len(success)} | ❌ {failed}")
    
    text = f"✅ Готово!\n\nКуплено: {len(success)}/{qty}\nЦена: {price_per_one}₽/шт\nПотрачено: {round(price_per_one*len(success),2)}₽\n💰 Баланс: {user.balance:.2f}₽\n\n"
    if success:
        text += "\n".join(success[:10])
    
    builder = InlineKeyboardBuilder()
    builder.button(text="🛒 Ещё", callback_data="buy_imo")
    builder.button(text="🔙 Меню", callback_data="main_menu")
    builder.adjust(1)
    
    await callback.message.edit_text(text, reply_markup=builder.as_markup())
    await callback.answer()

@group_router.callback_query(F.data.startswith("g_"))
async def group_purchase(callback: CallbackQuery, session: AsyncSession):
    parts = callback.data.split("_")
    country = parts[1]
    group_id = int(parts[2])
    
    group = await session.get(Group, group_id)
    if not group:
        await callback.answer("Группа не подключена.")
        return
    
    owner = await session.get(User, group.owner_id)
    if not owner:
        await callback.answer("Владелец не найден.")
        return
    
    try:
        prices = await get_prices_cached()
        
        price_rub = None
        found_country = country
        
        if country == "any":
            best = min(prices, key=lambda x: x['price']) if prices else None
            if best:
                price_rub = round(best['price'] * get_markup(), 2)
                found_country = best['country']
        else:
            found = [p for p in prices if p['country'] == country]
            if found:
                best = min(found, key=lambda x: x['price'])
                price_rub = round(best['price'] * get_markup(), 2)
        
        if not price_rub:
            await callback.answer("Нет номеров.")
            return
        
        if owner.balance < price_rub:
            await callback.answer("Недостаточно средств у владельца.")
            return
        
        result = await smsfast_service.buy_number(country=found_country)
        
        if result.get('id') and 'error' not in result:
            order_id = result['id']
            phone = result['phone']
            
            for _ in range(30):
                await asyncio.sleep(2)
                code = await smsfast_service.get_sms(order_id)
                if code:
                    owner.balance -= price_rub
                    owner.total_spent += price_rub
                    owner.last_activity = datetime.now()
                    
                    session.add(Purchase(user_id=callback.from_user.id, country=found_country, phone=phone, sms_code=code, price_rub=price_rub, group_id=group_id, group_owner_id=owner.telegram_id))
                    session.add(Transaction(user_id=owner.telegram_id, type='purchase', amount=-price_rub, description=f'IMO {found_country} (группа)'))
                    await session.commit()
                    
                    await callback.message.edit_text(f"✅ Готово!\n\n📱 {phone}\n🌍 {found_country.upper()}\n💬 Код: {code}\n💰 Списано с владельца: {price_rub}₽")
                    await callback.answer()
                    return
            
            await smsfast_service.cancel_order(order_id)
            await callback.answer("SMS не получена.")
        else:
            await callback.answer("Ошибка покупки.")
    except Exception as e:
        await callback.answer(f"Ошибка: {str(e)}")
